æ€è·¯ï¼šç”»äº†ä¸‹å›¾ï¼Œå‡è®¾ä»å³ä¸‹è§’å¼€å§‹ï¼Œå¾€ä¸Šï¼Œå¾€å·¦æ‰©æ•£ï¼Œå¯ä»¥å‘ç°ï¼Œï¼ˆiï¼Œjï¼‰èƒ½æ„æˆå¤šå¤§çš„æ­£æ–¹å½¢ä¸å®ƒçš„å³ã€ä¸‹ä»¥åŠå³ä¸‹æœ‰å…³ï¼Œå› æ­¤ï¼Œè®¾ç½®è½¬ç§»æ–¹ç¨‹ï¼š

class Solution {
    public int maximalSquare(char[][] matrix) {
        int max = 0;
        int n = matrix.length-1, m = matrix[0].length-1;
        int[][] maxL = new int[n+1][m+1];
        for(int i=n;i>=0;i--){
            for(int j=m;j>=0;j--){
                if(i == n || j==m){
                    maxL[i][j] = matrix[i][j] == '1' ? 1 : 0;
                    max = Math.max(max,maxL[i][j]);
                    continue;
                }
                if(matrix[i][j] == '0') {
                    maxL[i][j] = 0;
                    continue;
                }
                int minTmp = Math.min(maxL[i+1][j],maxL[i][j+1]);
                minTmp = Math.min(maxL[i+1][j+1],minTmp);
                maxL[i][j] = minTmp+1;
                max = Math.max(max,maxL[i][j]);
            }
        }
        return max*max;
    }
}

        int[] dp = new int[n+1];
