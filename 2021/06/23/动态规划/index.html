class Solution {
    public int maximalSquare(char[][] matrix) {
        int max = 0;
        int n = matrix.length-1, m = matrix[0].length-1;
        int[][] maxL = new int[n+1][m+1];
        for(int i=n;i>=0;i--){
            for(int j=m;j>=0;j--){
                if(i == n || j==m){
                    maxL[i][j] = matrix[i][j] == '1' ? 1 : 0;
                    max = Math.max(max,maxL[i][j]);
                    continue;
                }
                if(matrix[i][j] == '0') {
                    maxL[i][j] = 0;
                    continue;
                }
                int minTmp = Math.min(maxL[i+1][j],maxL[i][j+1]);
                minTmp = Math.min(maxL[i+1][j+1],minTmp);
                maxL[i][j] = minTmp+1;
                max = Math.max(max,maxL[i][j]);
            }
        }
        return max*max;
    }
}

    public int rob(int[] nums) {
        int n = nums.length;
        if(n == 1)return nums[0];

        dp[0] = nums[0];

class Solution {
    public int rob(int[] nums) {
        int length = nums.length;
        if (length == 1) {
            return nums[0];
        } else if (length == 2) {
            return Math.max(nums[0], nums[1]);
        }
        return Math.max(robRange(nums, 0, length - 2), robRange(nums, 1, length - 1));
    }

    public int robRange(int[] nums, int start, int end) {
        int first = nums[start], second = Math.max(nums[start], nums[start + 1]);
        for (int i = start + 2; i <= end; i++) {
            int temp = second;
            second = Math.max(first + nums[i], second);
            first = temp;
        }
        return second;
    }
}

class Solution {
    HashMap<TreeNode,Integer> f;
    HashMap<TreeNode,Integer> g;
    public int rob(TreeNode root) {
        if(root == null)return 0;
        f = new HashMap<>();
        g = new HashMap<>();
        dfs(root);
        return Math.max(f.get(root),g.get(root));
    }
    public void dfs(TreeNode root){
        if(root == null) return;
        if(root.left == null && root.right == null){
            f.put(root,root.val); // é€‰æ‹©è¯¥node
            g.put(root,0); // ä¸é€‰æ‹©è¯¥node
            return;
        }
        dfs(root.left);
        dfs(root.right);
        int gl = g.getOrDefault(root.left,0), gr = g.getOrDefault(root.right,0);
        int fl = f.getOrDefault(root.left,0), fr = f.getOrDefault(root.right,0);
        f.put(root,root.val + gl+gr);
        g.put(root,Math.max(fl,gl) + Math.max(fr,gr));
    }
}

class Solution {
    public int tribonacci(int n) {
        if(n==0||n==1)return n;
        if(n==2) return 1;
        int[] dp = new int[n+1];
        dp[0] = 0;
        dp[1] = 1;
        dp[2] = 1;
        for(int i=3;i<=n;i++){
            dp[i] = dp[i-1] + dp[i-2] + dp[i-3];
        }
        return dp[n];
    }
}

class Solution {
    private Map<String, Integer> map;
    public int numDecodings(String s) {
        if(s.length() == 0) return 0;
        map = new HashMap<>();
        return dfs(s, 0, s.length());
    }
    public int dfs(String num, int start, int end) {
        int res = 0;
        if(start >= end) return 1;
        int i = start;
        char ch = num.charAt(i);
        if(ch == '0') return 0;
        if(ch == '1' && i + 1 < end ) {
            String sub = num.substring(start+1,end);
            String sub2 = num.substring(start+2, end);
            if(null != map.get(sub)) {
                res += map.get(sub);
            } else {
                 res += dfs(num, start + 1, end);
            }
            if(null != map.get(sub2)) {
                res += map.get(sub2);
            } else {
                 res += dfs(num, start + 2, end);
            }           
        } else if (ch == '2' &&  i + 1 < end ) {
            char ch2 = num.charAt(i + 1);
            String sub = num.substring(start+1,end);
            if(null != map.get(sub)) {
                res += map.get(sub);
            } else {
                 res += dfs(num, start + 1, end);
            }
            if ( ch2 <= '6') {
                String sub2 = num.substring(start+2, end);
                if(null != map.get(sub2)) {
                    res += map.get(sub2);
                } else {
                    res += dfs(num, start + 2, end);
                } 
            }
        } else {
            String sub = num.substring(start+1,end);
            if(null != map.get(sub)) {
                res += map.get(sub);
            } else {
                 res += dfs(num, start + 1, end);
            }
        }
        String sub = num.substring(start,end);
        map.put(sub, res);
        return res;
    }
}

class Solution {
    public int numDecodings(String s) {
        int n = s.length();
        int[] f = new int[n+1];
        f[0] = 1;
        for(int i=1;i<=n;i++) {
            char ch = s.charAt(i-1); // 
            if( ch != '0') {
                f[i] += f[i - 1];
            }
            if (i > 1) {
                char ch2 = s.charAt(i-2);
                if (ch2 != '0' && ((ch2 - '0') * 10 + (ch - '0')) <= 26) {
                    f[i] += f[i-2];
                }
            }
        }
        return f[n];
    }
}
