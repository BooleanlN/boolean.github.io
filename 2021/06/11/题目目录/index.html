146. LRU 缓存机制

class LRUCache {
    private Node head;
    private Node tail;
    private int capacity;
    private Map<Integer, Node> cache;
    class Node{
        Node prev;
        Node next;
        int val;
        int key;
        Node(int key, int value){
            this.val = value;
            this.key = key;
        }
        Node(){

        }
    }
    public LRUCache(int capacity) {
        cache = new HashMap<>();
        head = new Node();
        tail = new Node();
        head.next = tail;
        tail.prev = head;
        this.capacity = capacity;
    }
    
    public int get(int key) {
        if(!cache.contains(key)) {
            return -1;
        }
        Node node = cache.get(key);
        moveToHead(node);
        return node.val;
    }
    public void moveToHead(Node node) {
        removeNode(node);
        Node tmp = head.next;
        head.next = node;
        node.next = tmp;
        tmp.prev = node;
        node.prev = head;
    }
    public void removeNode(Node node){
        node.prev.next = node.next;
        node.next.prev = node.prev;
    }
    public void removeTail(){
        Node tmp = tail.prev;
        tail.prev = tmp.prev;
        tmp.prev.next = tail;
    }
    public void put(int key, int value) {
        if(cache.size() > capacity){
            removeTail();
        }
        if(cache.contains(key)) {
            Node node = cache.get(key);
            node.val = value;
            moveToHead(node);
            return;
        }
        Node node = new Node(key,value);
        moveToHead(node);
        cache.put(key,node);
    }
}
