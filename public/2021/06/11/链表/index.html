public class Solution {
    public ListNode getIntersectionNode(ListNode headA, ListNode headB) {
        if(headA == null || headB == null) return null;
        ListNode p = headA, q = headB;
        int m = 0,n = 0;
        while(p != null){
            p = p.next;
            m ++;
        }
        while(q != null){
            q = q.next;
            n ++;
        }
        p = headA;
        q = headB;
        int c = 0;
        while(p != q){
            p = p.next;
            if(p == null){
                p = headB;
            }
            q = q.next;
            if(q == null){
                q = headA;
            }
            c ++;
            if(c >= m + n){ // è¶…å‡ºæ—¶ï¼Œè¡¨ç¤ºä¸¤è€…æ— äº¤ç‚¹

class Solution {
    public ListNode addTwoNumbers(ListNode l1, ListNode l2) {
        Deque<ListNode> stack1 = new LinkedList<>();
        Deque<ListNode> stack2 = new LinkedList<>();
        ListNode res = new ListNode(-1);
        while(l1 != null) {
            stack1.push(l1);
            l1 = l1.next;
        }
        while(l2 != null) {
            stack2.push(l2);
            l2 = l2.next;
        }
        int cnt = 0;
        ListNode p = res;
        while(!stack1.isEmpty() && !stack2.isEmpty()){
            int a = stack1.pop().val, b = stack2.pop().val;
            int sum = cnt + a + b;
            cnt = sum / 10;
            sum %= 10;
            p = res.next;
            res.next = new ListNode(sum);
            res.next.next = p;
        }
        while(!stack1.isEmpty()) {
            int a = stack1.pop().val;
            int sum = cnt + a;
            cnt = sum / 10;
            sum %= 10;
            p = res.next;
            res.next = new ListNode(sum);
            res.next.next = p;
        }
        while(!stack2.isEmpty()) {
            int a = stack2.pop().val;
            int sum = cnt + a;
            cnt = sum / 10;
            sum %= 10;
            p = res.next;
            res.next = new ListNode(sum);
            res.next.next = p;
        }
        if(cnt != 0) {
            p = res.next;
            res.next = new ListNode(cnt);
            res.next.next = p;
        }
        return res.next;
    }
}
