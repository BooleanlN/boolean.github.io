    class Solution {
    public int numSquares(int n) {
        int[] dp = new int[n+1];
        dp[0] = 0;
        dp[1] = 1;
        for(int i=2;i<=n;i++){
            int tmp = 1;
            dp[i] = 10000;
            while(tmp*tmp <= i){
                if (tmp*tmp == i){
                    dp[i] = 1;
                    break;
                }
                dp[i] = Math.min(dp[i],1 + dp[i-tmp*tmp]);
                tmp++;
            }
        }
        return dp[n];
    }
}

    class Solution {
    ArrayList<Integer> nums;
    public int numSquares(int n) {
        nums = new ArrayList<>();
        for(int i=1;i*i<=n;i++){
            nums.add(i*i);
        }
        for(int i=1;i<=n+1;i++){
            if(isValid(n,i)) return i;
        }
        return -1;
    }
    public boolean isValid(int n,int count){
        if(count == 1){
                return nums.contains(n);
        }
        for(int num:nums){
            if(isValid(n-num,count-1)){
                return true;
            }
        }
        return false;
    }
}
